int binary_search(int arr[], int size, int key){
    int lo = 0, hi = size - 1, mid;
    while(lo <= hi){
        mid = (lo + hi) >> 1;
        if(key == arr[mid]) return mid;
        if(key > arr[mid]) lo = mid + 1;
        else hi = mid - 1;
    }
    return -1;
}
